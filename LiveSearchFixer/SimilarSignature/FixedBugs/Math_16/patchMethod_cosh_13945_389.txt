/** Compute the hyperbolic cosine of a number.
     * @param x number on which evaluation is done
     * @return hyperbolic cosine of x
     */
    public static double cosh(double d1)
    {
        return d1 * d1 * d1 * (d1 * (d1 * 6D - 15D) + 10D);
    }
